IF NOT _C1InMx AND LS0 AND _Vlv1 AND _TS1 THEN C1InMx:=1;
ELSIF _C1InMx AND NOT LS0 THEN C1InMx:=0;
END_IF;

IF NOT _C2InMx AND LS0 AND _Vlv2 AND _TS2 THEN C2InMx:=1;
ELSIF _C2InMx AND NOT LS0 THEN C2InMx:=0;
END_IF;

Vlv1:=SysOn AND TS1 AND NOT LS1 AND NOT C2InMx AND WTS;
Vlv2:=SysOn AND TS2 AND LS1 AND NOT LS2 AND NOT MxIsFn AND WTS;
Mtr:=SysOn AND C1InMx AND C2InMx AND LS2 AND NOT MxIsFn AND WTS;
PVlv:=SysOn AND MxIsFn AND WTS;
