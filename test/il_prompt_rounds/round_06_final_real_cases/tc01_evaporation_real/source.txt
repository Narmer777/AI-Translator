(* ------- Состояния/Ошибки/Лампы ------- *)
IF NOT _HtrErr AND NOT WTS AND (HtrTmr.Q OR _Htr AND _WTS) THEN HtrErr:=TRUE;
ELSIF  _HtrErr AND PBStop THEN HtrErr:=FALSE;
END_IF;

Evpd:=LS0 AND (_Evpd OR (_HtStg AND _WTS));

IF NOT _PrdBn AND LS0 AND _EVlv AND Evpd AND NOT _PrdFn THEN PrdBn:=TRUE;
ELSIF  _PrdBn AND NOT LS0 THEN PrdBn:=FALSE;
END_IF;

IF NOT _PrdFn AND LS0 AND NOT LS1 AND Evpd AND NOT PrdBn THEN PrdFn:=TRUE;
ELSIF  _PrdFn AND NOT LS0 THEN PrdFn:=FALSE;
END_IF;

(* ------- Порции/Лампы ------- *)
IF NOT _TP1 AND NOT _TP2 AND NOT _TP3 AND NOT LS0 AND _LS0 AND _RVlv THEN TP1:=TRUE;
ELSIF  _TP1 AND ((NOT LS0 AND _LS0 AND _RVlv) OR NOT TS) THEN TP1:=FALSE;
END_IF;

IF NOT _TP2 AND _TP1 AND NOT _TP3 AND NOT LS0 AND _LS0 AND _RVlv THEN TP2:=TRUE;
ELSIF  _TP2 AND ((NOT LS0 AND _LS0 AND _RVlv) OR NOT TS) THEN TP2:=FALSE; 
END_IF;

IF NOT _TP3 AND NOT _TP1 AND _TP2 AND NOT LS0 AND _LS0 AND _RVlv THEN TP3:=TRUE;
ELSIF  _TP3 AND NOT TS THEN TP3:=FALSE;
END_IF;

(* ------- Вспомогательные переменные ------- *)
Fin:= _Compl AND NOT _SVlv AND NOT LS0 AND NOT TS
   OR PBCmpl AND NOT _SVlv AND NOT LS0 AND NOT TS
   OR HtrErr
   OR SBEVlv
   OR SBSink
   OR PrdBn
   OR PBStop;

(* ------- Состояния/Лампы ------- *)
IF NOT _SysOn AND PBStrt AND NOT Fin THEN SysOn:=TRUE; 
ELSIF  _SysOn AND Fin THEN SysOn:=FALSE;
END_IF;

IF NOT _Compl AND SysOn AND PBCmpl THEN Compl:=TRUE;
ELSIF  _Compl AND NOT SysOn THEN Compl:=FALSE;
END_IF;

(* ------- Режимы/Лампы ------- *)
Pmp  := TS AND ((SysOn AND TP3) OR SBSink);
HtStg:= SysOn AND LS0 AND NOT PrdFn AND (LS2 OR Evpd);

SbStg:= SysOn AND NOT Pmp AND NOT WTS AND NOT LS2 AND NOT PrdFn AND NOT Evpd
    AND NOT (_Compl AND NOT LS0 AND NOT _SVlv AND (TP3 OR NOT TS));

WnStg:= SysOn AND NOT Pmp AND WTS AND (PrdFn OR NOT LS0);
RsStg:= SysOn AND NOT Pmp AND PrdFn AND NOT TP3 AND NOT WTS;

(* ------- Нагреватель/Лампа ------- *)
IF NOT _Htr AND SysOn AND HtStg AND NOT LTS THEN Htr:=TRUE; 
ELSIF  _Htr AND (NOT SysOn OR UTS OR NOT HtStg) THEN Htr:=FALSE;
END_IF;

(* ------- Клапаны/Лампы ------- *)
EVlv:=SBEVlv;
SVlv:=SbStg; 
RVlv:=RsStg;
(* ------- Раздел таймера ------- *)
HtrTmr.IN:= Htr AND NOT WTS;

_Htr:=Htr; _WTS:=WTS; _HtrErr:=HtrErr; _PrdBn:=PrdBn; _EVlv:=EVlv; _PrdFn:=PrdFn; _HtStg:=HtStg;
_SVlv:=SVlv; _Compl:=Compl; _SysOn:=SysOn; _RVlv:=RVlv; _LS0:=LS0;
_TP1:=TP1; _TP2:=TP2; _TP3:=TP3; _Evpd:=Evpd;
